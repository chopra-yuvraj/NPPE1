## Overview

This notebook implements a **two-phase fine-tuning pipeline** for multi-label chest X-ray classification across 20 thoracic pathology classes using a **ViT-Large/16** backbone pretrained on ImageNet-21k. The scoring metric is an asymmetric cost function penalising false negatives (missed disease) five times more than false positives.

Key design choices:
- **ViT-Large/16 (224×224)** global attention captures long-range spatial relationships across the X-ray
- **Two-phase training** head warmup (frozen backbone) followed by full fine-tune with differential LRs
- **Mixup + Label Smoothing** regularisation to counter heavy class imbalance
- **Test-Time Augmentation (TTA × 7)** averaging stochastic augmentations at inference for robustness

```python
class XRayDataset(Dataset):
    """
    PyTorch Dataset for chest X-ray images.

    Args:
        df          : DataFrame with columns ['id', *class_names]
        img_dir     : Root directory containing image files
        transform   : torchvision transform pipeline
        class_names : Ordered list of class label column names
    """

    def __init__(self, df, img_dir, transform, class_names):
        self.df          = df
        self.img_dir     = img_dir
        self.transform   = transform
        self.class_names = class_names

    def __len__(self):
        return len(self.df)

    def __getitem__(self, idx):
        row   = self.df.iloc[idx]
        path  = os.path.join(self.img_dir, row['id'])
        img   = Image.open(path).convert('RGB')          # force 3-channel
        img   = self.transform(img)
        label = torch.tensor(row[self.class_names].values.astype(np.float32))
        return img, label
```

## 7. Model Architecture ViT-Large XRay

```python
class ViTLargeXRay(nn.Module):
    """
    ViT-Large/16 backbone with a custom multi-layer classification head.

    Architecture:
        Backbone : vit_large_patch16_224 (ImageNet-21k pretrained via timm)
                   307M parameters, 16×16 patch tokens, 1024-dim CLS output
        Head     : Linear(1024 → 512) → BN → GELU → Dropout(0.4) → Linear(512 → 20)

    The large receptive field of ViT-Large is well-suited for chest X-rays
    where pathology can appear anywhere and requires global context to detect.
    """

    def __init__(self, num_classes=NUM_CLASSES):
        super().__init__()
        # Load ViT-Large without the default classification head (num_classes=0)
        base          = timm.create_model(
            'vit_large_patch16_224', pretrained=True, num_classes=0
        )
        self.backbone = base
        in_features   = base.num_features          # 1024 for ViT-Large

        # Custom head with bottleneck, BN, and dropout for regularisation
        self.head = nn.Sequential(
            nn.Linear(in_features, 512),
            nn.BatchNorm1d(512),
            nn.GELU(),
            nn.Dropout(0.4),
            nn.Linear(512, num_classes)
        )

    def forward(self, x):
        x = self.backbone(x)    # (B, 1024) CLS token representation
        return self.head(x)     # (B, num_classes) Raw logits


model = ViTLargeXRay().to(DEVICE)

total = sum(p.numel() for p in model.parameters())
head  = sum(p.numel() for p in model.head.parameters())

print(f'ViTLargeXRay on {DEVICE}')
print(f'Total params   : {total:,}')
print(f'Head params    : {head:,}')
print(f'Backbone params: {back:,}')
```

## 9. Loss Function, Mixup, Label Smoothing & Training Loop

```python
# Binary cross-entropy for multi-label classification (one sigmoid per class)
criterion = nn.BCEWithLogitsLoss()
f1_metric = MultilabelF1Score(num_labels=NUM_CLASSES, average='macro').to(DEVICE)


def mixup_data(x, y, alpha=MIXUP_ALPHA):
    """
    Apply Mixup augmentation to a batch.

    Linearly interpolates between two randomly paired samples using a
    Beta(alpha, alpha) mixing coefficient. Helps smooth the decision boundary
    and reduces overfitting, particularly useful with scarce minority classes.

    Returns:
        x_mix : mixed images
        y_mix : mixed (soft) labels
    """
    lam   = np.random.beta(alpha, alpha)
    idx   = torch.randperm(x.size(0), device=x.device)
    x_mix = lam * x + (1 - lam) * x[idx]
    y_mix = lam * y + (1 - lam) * y[idx]
    return x_mix, y_mix


def smooth_labels(labels, smoothing=LABEL_SMOOTH):
    """
    Apply label smoothing to prevent overconfident predictions.

    Replaces hard 0/1 labels with softened values:
        1 → 1 - smoothing + smoothing/K
        0 → smoothing/K
    where K is the number of classes.
    """
    return labels * (1 - smoothing) + smoothing / NUM_CLASSES


def run_epoch(model, loader, optimizer, scheduler, train=True):
    """
    Run one full epoch of training or validation.

    During training:
      - Applies label smoothing to all labels
      - Applies Mixup with probability MIXUP_PROB
      - Clips gradients to norm 1.0 to stabilise training
      - Steps both optimizer and LR scheduler

    During validation:
      - Runs under torch.no_grad() for memory efficiency
      - Accumulates macro F1 across all batches

    Returns:
        avg_loss : average per-sample loss for the epoch
        f1_score : macro-averaged multilabel F1 score
    """
    model.train() if train else model.eval()
    f1_metric.reset()
    total_loss = 0
    n = 0

    ctx = torch.enable_grad() if train else torch.no_grad()
    with ctx:
        for imgs, labels in tqdm(loader, desc='train' if train else 'val'):
            imgs   = imgs.to(DEVICE)
            labels = labels.to(DEVICE)

            if train:
                # Regularise: smooth labels, then optionally mix pairs
                labels = smooth_labels(labels)
                if random.random() < MIXUP_PROB:
                    imgs, labels = mixup_data(imgs, labels)

            logits = model(imgs)
            loss   = criterion(logits, labels)

            if train:
                optimizer.zero_grad()
                loss.backward()
                torch.nn.utils.clip_grad_norm_(model.parameters(), 1.0)
                optimizer.step()
                scheduler.step()

            # Threshold at 0.5 for F1 computation
            preds = (torch.sigmoid(logits) > 0.5).int()
            f1_metric.update(preds, labels.int())
            total_loss += loss.item() * imgs.size(0)
            n += imgs.size(0)

    return total_loss / n, f1_metric.compute().item()
```

## 10. Phase 1 Head Warmup (Frozen Backbone)

```python
# Freeze all backbone parameters; only the classification head will train
for p in model.backbone.parameters():
    p.requires_grad = False

trainable = sum(p.numel() for p in model.parameters() if p.requires_grad)
print(f'Phase 1: {trainable:,} trainable params (head only)')

opt_p1 = torch.optim.AdamW(
    filter(lambda p: p.requires_grad, model.parameters()),
    lr           = LR_P1,
    weight_decay = WEIGHT_DECAY
)

total_steps_p1 = len(train_loader) * EPOCHS_P1
sch_p1 = get_cosine_schedule_with_warmup(
    opt_p1,
    num_warmup_steps   = total_steps_p1 // 10,   # 10% warmup
    num_training_steps = total_steps_p1
)

best_p1_f1   = 0
patience_cnt = 0

for epoch in range(EPOCHS_P1):
    tr_loss, tr_f1  = run_epoch(model, train_loader, opt_p1, sch_p1, train=True)
    va_loss, val_f1 = run_epoch(model, val_loader,   opt_p1, sch_p1, train=False)
    lr_now = sch_p1.get_last_lr()[0]

    print(f'P1 Ep {epoch+1}/{EPOCHS_P1} | '
          f'loss={tr_loss:.4f} train_f1={tr_f1:.4f} val_f1={val_f1:.4f} lr={lr_now:.2e}')

    if val_f1 > best_p1_f1:
        best_p1_f1   = val_f1
        patience_cnt = 0
        torch.save(model.state_dict(), '/kaggle/working/best_model_phase1.pth')
        print('  ✓ Saved best Phase 1 model')
    else:
        patience_cnt += 1
        if patience_cnt >= PATIENCE_P1:
            print('Early stopping Phase 1.')
            break

print(f'Phase 1 complete best val F1: {best_p1_f1:.4f}')
```

## 12. TTA Inference & Submission

```python
def predict_tta(model, img_id, n_tta=N_TTA):
    """
    Run Test-Time Augmentation for a single image.

    Applies the TTA transform pipeline n_tta times independently, runs each
    through the model, and averages the resulting probability vectors. This
    reduces variance in the final prediction without retraining.

    Args:
        model  : Trained ViTLargeXRay in eval mode
        img_id : Filename string (e.g. '00000001_000.png')
        n_tta  : Number of augmented passes to average

    Returns:
        probs : numpy array of shape (num_classes,) with averaged probabilities
    """
    path      = os.path.join(IMG_DIR, img_id)
    img       = Image.open(path).convert('RGB')
    all_probs = []

    for _ in range(n_tta):
        tensor = tta_tfms(img).unsqueeze(0).to(DEVICE)
        with torch.no_grad():
            logits = model(tensor)
            probs  = torch.sigmoid(logits).cpu().numpy()
            all_probs.append(probs)

    # Average across all TTA passes
    return np.mean(all_probs, axis=0)[0]


# Load the best checkpoint from Phase 2
model.load_state_dict(torch.load('/kaggle/working/best_model_phase2.pth'))
model.eval()

all_ids   = []
all_preds = []

for idx in tqdm(range(len(test_df)), desc=f'Inference (TTA x{N_TTA})'):
    img_id = test_df.iloc[idx]['id']
    probs  = predict_tta(model, img_id)

    # Hard argmax → single-label one-hot (required by submission format)
    pred    = int(np.argmax(probs))
    one_hot = [0] * NUM_CLASSES
    one_hot[pred] = 1

    all_ids.append(img_id)
    all_preds.append(one_hot)

# Build and validate the submission DataFrame
submission = pd.DataFrame(all_preds, columns=CLASS_NAMES)
submission.insert(0, 'id', all_ids)

assert (submission[CLASS_NAMES].sum(axis=1) == 1).all(), \
    'Row-sum check failed each row must have exactly one predicted class'
assert list(submission.columns) == list(sample_sub.columns), \
    'Column mismatch with sample_submission.csv'

submission.to_csv('/kaggle/working/submission.csv', index=False)
print(f'submission.csv saved {len(submission):,} rows')
print(submission[CLASS_NAMES].sum().sort_values(ascending=False))
```
